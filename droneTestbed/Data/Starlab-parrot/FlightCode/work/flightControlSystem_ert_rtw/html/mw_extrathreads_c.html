69void* keyboardserver_thread(void* i)



70{



71    struct sched_param param;



72    param.sched_priority = 9;



73    pthread_setschedparam(pthread_self(), 2, &param);



74    printf("keyboardserver_thread(): Entered the Keyboard Server thread... \n");



75    



76    if((sockfd = socket(AF_INET, SOCK_STREAM, 0)) < 0) {



77        printf("keyboardserver_thread(): ERROR : Could not create socket \n");



78        exit(1);



79    }



80    



81    printf("keyboardserver_thread(): Successfully created socket... \n");



82    



83    memset(&serv_addr, '0', sizeof(serv_addr));



84    



85    serv_addr.sin_family = AF_INET;



86    serv_addr.sin_addr.s_addr = htonl(INADDR_ANY);



87    serv_addr.sin_port = htons(12391);



88    



89    printf("keyboardserver_thread(): Bind socket... \n");



90    



91    // Fix : We could not run the Spider Flight Continuously. We need to wait for some time before you start the next flight.



92    // To fix the issue we are creating Socket with REUSEADDR.



93    if (setsockopt(sockfd, SOL_SOCKET, SO_REUSEADDR, &yes, sizeof(int)) < 0)



94    {



95        printf("keyboardserver_thread(): ERROR: Failure in Reusing address.. with error: %d\n",errno);



96        exit(1);



97    }



98    



99    



100    if(bind(sockfd, (struct sockaddr *)&serv_addr, sizeof(serv_addr)) < 0) {



101        printf("keyboardserver_thread(): ERROR Connection to reference value server failed %d \n", errno);



102        exit(1);



103    }



104    else



105    {



106        printf("keyboardserver_thread(): Bind completed .. listening to the socket !\n");



107        



108        if (listen(sockfd, 10)<0)



109        {



110            printf("keyboardserver_thread(): Socket error (listen) %d\n",errno);



111            exit(1);



112        }



113        connfd = accept(sockfd, (struct sockaddr*)NULL, NULL);



114        setsockopt(connfd, SOL_SOCKET, SO_REUSEADDR, &yes, sizeof(yes));



115        setsockopt(connfd, SOL_SOCKET, SO_SNDBUF, &buff_size, sizeof(buff_size));



116        clientConnect = 1;



117        printf("keyboardserver_thread(): Received connection from the host! \n");



118    }



119    



120   ///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////// 



121    



122     if((sockfd_GUI = socket(AF_INET, SOCK_STREAM, 0)) < 0) {



123        printf("keyboardserver_thread(): ERROR : Could not create GUI socket\n");



124        exit(1);



125    }



126    



127    printf("keyboardserver_thread(): Successfully created GUI socket... \n");



128    



129    memset(&serv_addr, '0', sizeof(serv_addr));



130    



131    serv_addr.sin_family = AF_INET;



132    serv_addr.sin_addr.s_addr = htonl(INADDR_ANY);



133    serv_addr.sin_port = htons(24099);



134    



135    printf("keyboardserver_thread(): Bind GUI socket...\n");



136    



137    // Fix : We could not run the Spider Flight Continuously. We need to wait for some time before you start the next flight.



138    // To fix the issue we are creating Socket with REUSEADDR.



139    if (setsockopt(sockfd_GUI, SOL_SOCKET, SO_REUSEADDR, &yes, sizeof(int)) < 0)



140    {



141        printf("keyboardserver_thread(): ERROR: Failure in Reusing address in GUI.. with error: %d\n",errno);



142        exit(1);



143    }



144    



145    



146    if(bind(sockfd_GUI, (struct sockaddr *)&serv_addr, sizeof(serv_addr)) < 0) {



147        printf("keyboardserver_thread(): ERROR Connection to GUI server failed %d \n", errno);



148        exit(1);



149    }



150    else



151    {



152        printf("keyboardserver_thread(): Bind completed .. listening to the GUI socket !\n");



153        



154        if (listen(sockfd_GUI, 10)<0)



155        {



156            printf("keyboardserver_thread(): GUI Socket error (listen) %d\n",errno);



157            exit(1);



158        }



159        connfd_GUI = accept(sockfd_GUI, (struct sockaddr*)NULL, NULL);



160        setsockopt(connfd_GUI, SOL_SOCKET, SO_REUSEADDR, &yes, sizeof(yes));



161        setsockopt(connfd_GUI, SOL_SOCKET, SO_SNDBUF, &buff_size, sizeof(buff_size));



162        printf("keyboardserver_thread(): Received connection to GUI from the host! \n");



163    }



164    



165   ////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////



166    return NULL;



167}

169void create_referencevalserver_thread(void)



170{



171    int err;



172    pthread_t img_tid;



173    int policy;



174    struct sched_param param;



175    pthread_attr_t attr;



176    size_t stacksize;



177    



178    pthread_attr_init(&attr);



179    pthread_attr_getstacksize(&attr,&stacksize);



180    err = pthread_attr_setstacksize(&attr,PTHREAD_STACK_MIN);



181    printf("create_referencevalserver_thread(): stack size before reference value thread creation %d err=%d\n", stacksize,err);



182    



183    img_tid = pthread_self();



184    printf("create_referencevalserver_thread(): reference value server thread id = %d \n",(int) img_tid);



185    pthread_getschedparam(img_tid, &policy,&param);



186    printf("create_referencevalserver_thread(): reference value server  policy = %d pri=%d\n", (int)policy,param.sched_priority);



187    



188    // param1.sched_priority = param.sched_priority - 1;



189    // pthread_setschedparam(tid,policy,&param1);



190    err = pthread_create(&tid, &attr, &keyboardserver_thread, NULL);



191    if (err != 0)



192        printf("create_referencevalserver_thread(): ERROR: can't create thread :[%s]", strerror(err));



193    else



194        printf("create_referencevalserver_thread(): Reference Server Thread created successfully\n");



195    



196    pthread_attr_getstacksize(&attr,&stacksize);



197    printf("create_referencevalserver_thread(): stacksize after reference thread creation %d \n", stacksize);



198}

207void* extmode_thread(void* i)



208{



209    static int counter = 0;



210    counter++;



211    



212    if (counter == 1)



213    {



214        int policy;



215        struct sched_param param ;



216        



217        pthread_getschedparam(pthread_self(), &policy,&param);



218        printf("extmode_thread(): External mode thread policy = %d and priority=%d\n", (int)policy,param.sched_priority);



219        



220        /* initialize external mode */



221        rtParseArgsForExtMode(0, NULL);



222        rtmSetErrorStatus(MW_StringifyDefineFunction(MODEL,_M), 0);



223        MW_StringifyDefineFunction(MODEL,_initialize)();



224        /* External mode */



225        rtSetTFinalForExtMode(&rtmGetTFinal(MW_StringifyDefineFunction(MODEL,_M)));



226        rtExtModeCheckInit(1);



227        rtERTExtModeStartMsg();



228    }



229    



230    while(runModel)



231    {



232        boolean_T rtmStopReq = false;



233        #ifdef MW_HAS_IMAGE_PROCESSING



234            rtExtModeOneStep(MW_StringifyDefineFunction(MODEL,_M->extModeInfo), 2, &rtmStopReq); // This is used when the Image processing is enabled along with multitasking. This will upload the step0() and step1() data.



235        #else



236            rtExtModeOneStep(MW_StringifyDefineFunction(MODEL,_M->extModeInfo), 1, &rtmStopReq); // This is used when only the control code is running. Multitasking is not enabled in this case and only the step() data is required to be uploaded.



237        #endif 



238        if (rtmStopReq)



239        {



240            printf("extmode_thread(): Thread received STOP command \n");



241            rtmSetStopRequested(MW_StringifyDefineFunction(MODEL,_M), true);



242            // Reason for the following code:



243            // If the user presses stop button before starting the flight code i.e before starting



244            // the execution step code.



245            rtExtModeShutdown(1);



246            MW_StringifyDefineFunction(MODEL,_terminate)();



247            runModel = 0;



248        }



249        



250        if (run_flag == 0)



251        {



252            rtExtModeShutdown(1);



253            MW_StringifyDefineFunction(MODEL,_terminate)();



254            runModel = 0;



255        }



256    }



257    



258    if (runModel == 0)



259    {



260        rtExtModeShutdown(1);



261        MW_StringifyDefineFunction(MODEL,_terminate)();



262    }



263    



264    background_task_completed = true;



265    return NULL;



266}

268void create_background_thread(void)



269{



270    int err, status;



271    pthread_attr_t attr;



272    size_t stacksize;



273    struct sched_param param;



274    



275    pthread_attr_init(&attr);



276    err = pthread_attr_setinheritsched(&attr,PTHREAD_EXPLICIT_SCHED);



277    printf("create_background_thread(): ext thread inherit scheduler option error=%d\n", err);



278    



279    status = pthread_attr_setdetachstate(&attr, PTHREAD_CREATE_DETACHED);



280    printf("create_background_thread(): value of detach status=%d\n", status);



281    



282    err = pthread_attr_setstacksize(&attr,PTHREAD_STACK_MIN);



283    printf("create_background_thread(): stacksize before thread creation err=%d\n", err);



284    



285    /* In case of Mambo , the policy of rsedu_control.c is set as 2 (SCHED_RR) and priority 32



286     * In case of Rolling Spider , the policy of rsedu_control.c is set as 0 (SCHED_OTHER) and priority 0



287     * Setting the policy and priority of External Mode thread as 0 (SCHED_OTHER) and priority 0 */



288    



289    status = pthread_attr_setschedpolicy(&attr, SCHED_OTHER);



290    printf("create_background_thread(): value of scheduler policy status=%d\n", status);



291



292    param.sched_priority = 0;



293    status = pthread_attr_setschedparam(&attr, &param);



294    printf("create_background_thread(): value of scheduler param status=%d\n", status);



295    



296    err = pthread_create(&ext_tid, &attr, &extmode_thread, NULL);



297    if (err != 0)



298        printf("create_background_thread(): can't extmode thread :[%s]", strerror(err));



299    else



300        printf("create_background_thread(): External mode Thread created successfully\n");



301    



302    pthread_attr_getstacksize(&attr,&stacksize);



303    printf("create_background_thread(): stacksize after ext thread creation %d \n", stacksize);



304    



305}
