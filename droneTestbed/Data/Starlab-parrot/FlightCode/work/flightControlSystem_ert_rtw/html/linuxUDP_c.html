81void CreateTargetUDPInterface(void *hl)



82{



83    HostLibrary *hostLib = (HostLibrary*)hl;



84    *hostLib->errorMessage = '\0';



85    memset(hostLib, 0, sizeof(HostLibrary));



86    



87    /* Indicate that the structure is valid for future function calls. */



88    hostLib->library      = (void*) 1;



89    



90    /* Set the function pointers to point to target UDP implementation. */



91    hostLib->libCreate    = (void*) TGT_libCreate;



92    hostLib->libStart     = (void*) TGT_libStart;



93    hostLib->libReset     = (void*) NULL;



94    hostLib->libUpdate    = (void*) TGT_libUpdate;



95    hostLib->libOutputs   = (void*) TGT_libOutputs;



96    hostLib->libTerminate = (void*) TGT_libTerminate;



97    hostLib->libDestroy   = (void*) TGT_libDestroy;



98}

101void DestroyTargetUDPInterface(void *hl)



102{



103    HostLibrary *hostLib = (HostLibrary*)hl;



104    if(hostLib->library) {



105        /* Clear the structure (except for any possible error message). */



106        memset(hostLib, 0, sizeof(HostLibrary) - MAX_ERR_MSG_LEN);



107    }



108}

111void TGT_libCreate(void **device_, char *err, NetworkInputOrOutput io,



112        const char *localURL, int localPort, const char *remoteURL,



113        int remotePort, int bufferSize, int dataTypeSize, int blockingTime)



114{



115    int flags;



116    struct timeval tv;



117    int ret;



118    TgtSockStructT *device;



119    /* verification of the local port*/



120    /* case it lower than zero, assign to zero so it can dynamically assigned */



121    if(localPort < 0) {



122        localPort = 0;



123    }



124    device = (TgtSockStructT *) malloc (sizeof(TgtSockStructT));



125    



126    



127    bzero(&device->remotesockaddr, sizeof(device->remotesockaddr));



128    device->remotesockaddr.sin_family = AF_INET;



129    device->remotesockaddr.sin_port = htons(remotePort);



130    



131    inet_aton(remoteURL, &device->remotesockaddr.sin_addr);



132    



133    bzero(&device->localsockaddr, sizeof(device->localsockaddr));



134    device->localsockaddr.sin_family = AF_INET;



135    device->localsockaddr.sin_port = htons(localPort);



136    inet_aton(localURL, &device->localsockaddr.sin_addr);



137    



138    device->sockfd = socket(AF_INET, SOCK_DGRAM, 0);



139    if ( device->sockfd < 0) {



140        printf ("error creating socket\n");



141    }



142    if ((device->remotesockaddr.sin_addr.s_addr == INADDR_BROADCAST) ||



143            (device->localsockaddr.sin_addr.s_addr == INADDR_BROADCAST)) {



144        static int broadcast_en = 1;



145        ret = setsockopt(((TgtSockStructT *) device)->sockfd, SOL_SOCKET,



146                SO_BROADCAST, &broadcast_en, sizeof(broadcast_en));



147        if (ret < 0) {



148            perror("ERROR in SO_BROADCAST");



149        }



150    }



151    



152    if ( io == NETWORK_INPUT) {



153        ret = setsockopt(((TgtSockStructT *) device)->sockfd, SOL_SOCKET,



154                SO_RCVBUF, &bufferSize, sizeof(bufferSize));



155    }



156    else if (io == NETWORK_OUTPUT) {



157        ret = setsockopt(((TgtSockStructT *) device)->sockfd, SOL_SOCKET,



158                SO_SNDBUF, &bufferSize, sizeof(bufferSize));



159    }



160    else {



161        ret  = -1;



162        perror("INVALID NetworkIOValue");



163    }



164    



165    if (ret < 0) {



166        perror("ERROR in setsockopt");



167    }



168    else{



169        int realSize = 0;



170        socklen_t len = sizeof(realSize);



171        



172        if (io == NETWORK_INPUT) {



173            getsockopt(((TgtSockStructT *) device)->sockfd,SOL_SOCKET,SO_RCVBUF,&realSize, &len);



174        }



175        else {



176            getsockopt(((TgtSockStructT *) device)->sockfd,SOL_SOCKET,SO_SNDBUF,&realSize, &len);



177        }



178        if (realSize < bufferSize) {



179            fprintf(stderr, "Buffer requested size: %d Bytes. Socket buffer limit: %d Bytes\n",bufferSize,realSize);



180        }



181    }



182    



183    if(blockingTime == 0) {



184        /* set non-blocking socket option */



185        flags = fcntl(((TgtSockStructT *) device)->sockfd, F_GETFL, 0);



186        fcntl(((TgtSockStructT *) device)->sockfd, F_SETFL, flags | O_NONBLOCK);



187    }



188    else if(blockingTime != MIN_int32_T) {



189        /* if not set to inf in the block mask */



190        tv.tv_sec  = (long) (blockingTime/1000);



191        tv.tv_usec = (long) (blockingTime % 1000);



192        setsockopt(((TgtSockStructT *) device)->sockfd, SOL_SOCKET,



193                SO_RCVTIMEO, &tv, sizeof(tv));



194    }



195    device->dataTypeSize = dataTypeSize;



196    *device_ = device;



197    err[0] = '\0';



198}

201void TGT_libStart(void *device, char *err)



202{



203    if(device) {



204        bind(((TgtSockStructT *) device)->sockfd,



205                (struct sockaddr *) &((TgtSockStructT *) device)->localsockaddr,



206                sizeof(((TgtSockStructT *) device)->localsockaddr));



207    }



208}

211void TGT_libUpdate(void *device, char *err, const void *src, int nSamples)



212{



213    int n;



214    



215    if(device) {



216        n = sendto(((TgtSockStructT *) device)->sockfd, src,



217                ((TgtSockStructT *) device)->dataTypeSize * nSamples, 0,



218                (struct sockaddr *) &((TgtSockStructT *) device)->remotesockaddr,



219                sizeof(((TgtSockStructT *) device)->remotesockaddr));



220        /* if(n<0)



221         * printf("error writing to socket\n");



222         */



223    }



224}



225



226



227void TGT_libOutputs(void *device, char *err, void *src, int *nSamples)



228{



229    int n = 0;



230    



231    if(device) {



232        n = recvfrom(((TgtSockStructT *) device)->sockfd, src,



233                ((TgtSockStructT *) device)->dataTypeSize * (*nSamples),



234                0, NULL, NULL);



235        if (n > 0) {



236            *nSamples = n; /* Return number of bytes received */



237        }



238        else {



239            *nSamples = 0; /* No UDP packets received */



240        }



241    }



242}
