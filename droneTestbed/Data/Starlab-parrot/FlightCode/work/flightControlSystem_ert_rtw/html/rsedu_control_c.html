52//! Amount of take off cycles.

59

77/**



78 * @brief The function RSEDU_control is called at 200Hz to generate reference motor commands based on sensor values



79 *



80 * @param[in] hal_sensors_data Structure containing sensors current values



81 * @param[out] hal_sensors_cmd Structure containing commands to send to motors and LEDs



82 */



83void RSEDU_control(HAL_acquisition_t* hal_sensors_data, HAL_command_t* hal_sensors_cmd)



84{



85    //flight phases



86    



87    //process control



88    static int counter = 0;



89    static int counter_noOF = 0;



90    // counter to wait for TCP connection request from host



91    static int waitForTCPCounter = 0;



92    



93    // counter to wait for start command from host



94    static int waitForStartCounter = 0;



95    



96    static float MIN_BATTTAKEOFF        = 50.0; // Minimum battery level for take-off



97    static float MIN_BATT               = 30.0; // Minimum battery for continuing flight



98    static int MAX_noOF                 = 50;   // Maximum acceptable numbers of cycles without optical flow



99    



100    //power/usr input



101    static double powerGain = 0;



102    



103    //data handling



104    static double battLevelAvg;



105    



106    //float ofDefined;



107    static bool pressureSensorOk = false;



108    static float ofQuality;



109    



110#if (1 == MW_CONFIG_ENABLEPROFILING)



111    //ptiming - declare and start



112    //------------



113    long long start = 0;



114    static FILE *ptfile;



115#endif



116    



117    //------------



118    



119    // Create shorter aliases for the function arguments



120    HAL_acquisition_t* in = hal_sensors_data;



121    HAL_command_t    * out = hal_sensors_cmd;



122    



123    // Create the necessary Model Specific Buses (data structures) for model run



124    



125    static real32_T calibrationData[8] = {0.0}; // data to be passed to the Model for calib data



126    static CommandBus *keyboardServerCommands = &cmd_inport; // defined in flightControlSystem_types.h



127    static SensorsBus *SensorData = &sensor_inport; // defined in flightControlSystem_types.h



128    



129    //ABORT FLIGHT and exit if run_flag 0



130    if (run_flag == 0)



131    {



132        stop_flight(out);



133#ifdef EXT_MODE



134        if (background_task_completed == false)



135        {



136            return;



137        }



138#endif



139        



140        if (counter > calibCycles)



141        {



142#ifdef MATFILE_LOGGING



143            printf("rsedu_control():Saving logged data after %i cycles... \n", counter);



144            rt_StopDataLogging(MATFILE, MW_StringifyDefineFunction(MODEL,_M)->rtwLogInfo);



145            printf("rsedu_control(): Saving logged data... DONE \n");



146#endif



147        }



148        



149        // Close the optical flow FIFO



150        if (1 == MW_CONFIG_ENABLEOPTICALFLOW)



151        {



152            close(of_fifo);



153        }



154        



155#if (1 == MW_CONFIG_ENABLEPROFILING)



156        fclose(ptfile);



157#endif



158  



159        printf("rsedu_control(): Flight Completed after %i cycles ... DONE \n", counter);



160        



161#ifdef MW_HAS_IMAGE_PROCESSING



162        printf("RSEDU_image_processing: Average image processing time is %lldms \n", avrgIpTime);



163#endif



164        MW_StringifyDefineFunction(MODEL,_terminate)();



165        write(connfd_GUI, "De", 2);



166        usleep(1000);



167        printf("rsedu_control(): Good night! \n");



168        exit(0);



169    }



170    



171    //Process Control



172    counter++;



173#if (1 == MW_CONFIG_ENABLEPROFILING)



174    ptimer_start(&start);



175#endif



176    //==================



177    //Flight Stages s0-4



178    //==================



179    



180    //s0: Initialize (server connection, get user settings)



181    // ----------------------------------------------------



182    if (counter == 1)



183    {



184        pthread_t img_tid;



185        int policy;



186        struct sched_param param;



187        



188        img_tid = pthread_self();



189        printf("rsedu_control(): Control processing thread id = %d \n",(int) img_tid);



190        pthread_getschedparam(img_tid, &policy,&param);



191        printf("rsedu_control(): Control thread policy = %d pri=%d\n", (int)policy,param.sched_priority);



192        



193        printf("rsedu_control(): Battery output voltage Integrated: %5.2f V - %0d percents\n", in->HAL_vbat_SI.vbat_V, (int)in->HAL_vbat_SI.vbat_percentage);



194        printf("rsedu_control(): used: %d, users: %d, gyrotemp %f, acctemp %f, presstmp %f \n", (int)in->used, (int)in->count_user, in->HAL_gyro_SI.temperature, in->HAL_acc_SI.temperature, in->HAL_pressure_SI.temperature);



195        // Initialize the timer for communicating with the drone



196#if (1 == MW_CONFIG_ENABLEPROFILING)



197        ptimer_init(__func__, &(ptfile), &run_flag);



198#endif



199    }



200    



201    //s0: Comms and Model (Initialize inter-thread communication, model and calibration data)



202    // ----------------------------------------------------



203    else if (counter == 2)



204    {



205#ifdef MAMBO



206        out->HAL_ultrasound_command = ULTRASOUND_CMD_START;



207#endif



208        // Connect to Optical Flow



209        if (1 == MW_CONFIG_ENABLEOPTICALFLOW)



210        {



211            if (open_optical_flow_comm() == RETURN_FAIL)



212            {



213                printf("rsedu_control(): WARNING optical flow might not be running, %d!\n\n", of_fifo);



214            }



215            else



216            {



217                printf("rsedu_control(): Got optical flow connection, %d!\n", of_fifo);



218            }



219        }



220        else



221        {



222            printf("rsedu_control(): Optical Flow deactivated! \n");



223        }



224    }



225    //s0: Wait for the reference value server connection and also wait for the takeoff command



226    else if (counter == 3)



227    {



228        //keep both fifos empty (Optical Flow and Vision)



229        if (1 == MW_CONFIG_ENABLEOPTICALFLOW)



230        {



231            read(of_fifo, (float*)(&of_data), sizeof(of_data));



232        }



233        



234        if (clientConnect == 0)



235        {



236            if (waitForTCPCounter == 0)



237            {



238                create_referencevalserver_thread();



239            }



240#ifdef EXT_MODE



241            if (waitForTCPCounter == 1)



242            {



243                // Create background thread only when external mode is enabled



244                create_background_thread();



245            }



246#endif



247            /*Print on droneFlightlog every 1 second*/



248            if (waitForTCPCounter % 200 == 0)



249            {



250                printf("rsedu_control(): Waiting for connection request from host! \n");



251                printf("\n\n **** Sensor Values **** \n");



252                printf("rsedu_control(): sonar altitude: %fmeters, pressure: %fpascal, gyrotemp: %fcelsius\n", in->HAL_ultrasound_SI.altitude, in->HAL_pressure_SI.pressure,in->HAL_gyro_SI.temperature);



253                printf("rsedu_control(): gyro x: %frad/s, y: %frad/s, z: %frad/s\n", in->HAL_gyro_SI.x,in->HAL_gyro_SI.y,in->HAL_gyro_SI.z);



254                printf("rsedu_control(): accel x: %fm/s2, y: %fm/s2, z: %fm/s2\n", in->HAL_acc_SI.x,in->HAL_acc_SI.y,in->HAL_acc_SI.z);



255                printf("rsedu_control(): Battery output voltage Integrated: %5.2fV - %0d percents\n", in->HAL_vbat_SI.vbat_V, (int)in->HAL_vbat_SI.vbat_percentage);



256            }



257            waitForTCPCounter ++;



258            counter = 2;



259            return;



260        }



261        else if (isMotorOn == 0)



262        {



263            /*Print on droneFlightlog every 1 second*/



264            if (waitForStartCounter % 200 == 0)



265            {



266                printf("rsedu_control(): Waiting for takeoff command from host! \n");



267            }



268            waitForStartCounter++;



269            /* Wait till start command is received. Do not start counter*/



270            counter = 2;



271            



272            /*Read if start command has been received*/



273            fcntl(connfd, F_SETFL, O_NONBLOCK);



274            memset(recvBuff, '\0', sizeof(recvBuff));



275            recv(connfd, recvBuff, sizeof(recvBuff), O_NONBLOCK);



276            processRequest(connfd,recvBuff, in, out, keyboardServerCommands, &run_flag); //  function call to process data sent from MATLAB APIs



277            return;



278        }



279        



280        /* Set powerGain and flight duration values received from host */



281        onCycles = flightDuration * 200;



282        powerGain = throttle/100.0;



283        



284        /* Initialize model */



285        printf("rsedu_control(): Simulink model init started! \n");



286        



287        rtmSetErrorStatus(MW_StringifyDefineFunction(MODEL,_M), 0);



288        MW_StringifyDefineFunction(MODEL,_initialize)();



289        printf("rsedu_control(): Simulink model init completed! \n");



290        



291        /*Hardcoding MAT-File logging time to Flight time*/



292#ifdef MATFILE_LOGGING



293        rtmSetTFinal(MW_StringifyDefineFunction(MODEL,_M), flightDuration);



294#endif



295        



296        //init sensor calibration measurements



297        printf("rsedu_control(): Initializing Calibration Data! \n");



298        initialize_calibration_data(calibrationData, &battLevelAvg, in);



299        



300        //Activate motors



301        out->command = BLDC_CMD_START;



302        printf("rsedu_control(): Drone Motors Activated! \n");



303        



304        //s1: Record calibration data



305        // --------------------------



306    }



307    else if (counter < calibCycles)



308    {



309        



310        if (counter%200 == 0)



311        {



312            printf("rsedu_control(): Calibrating Data . . . \n");



313        }



314        perform_calibration(counter, calibrationData,&battLevelAvg, in);



315        



316        if (in->HAL_pressure_SI.pressure != 0)



317        {



318            pressureSensorOk = true;



319        }



320        



321        //keep both fifos empty (Optical Flow and Vision)



322        if (1 == MW_CONFIG_ENABLEOPTICALFLOW)



323        {



324            read(of_fifo, (float*)(&of_data), sizeof(of_data));



325        }



326        



327        //Power motors with 0 velocity



328        stop_flight(out);



329        



330        return;



331        



332        //s2: Initialize dynamic model for control



333        // --------------------------------------



334    }



335    else if (counter == calibCycles)



336    {



337        printf("rsedu_control(): Batterylevel: %f\n", battLevelAvg);



338        



339        // Passing calibration data to Simulink



340        SensorData->SensorCalibration[0] = calibrationData[0];



341        SensorData->SensorCalibration[1] = calibrationData[1];



342        SensorData->SensorCalibration[2] = calibrationData[2];



343        SensorData->SensorCalibration[3] = calibrationData[3];



344        SensorData->SensorCalibration[4] = calibrationData[4];



345        SensorData->SensorCalibration[5] = calibrationData[5];



346        SensorData->SensorCalibration[6] = calibrationData[6];



347        SensorData->SensorCalibration[7] = calibrationData[7];



348        



349        printf("rsedu_control(): Sensorcal: %f :: %f :: %f :: %f :: %f :: %f :: %f :: %f \n",



350                calibrationData[0],



351                calibrationData[1],



352                calibrationData[2],



353                calibrationData[3],



354                calibrationData[4],



355                calibrationData[5],



356                calibrationData[6],



357                calibrationData[7]);



358        



359        //Stop if angled take-off



360        if ((!MW_CONFIG_NOSAFETY) && fabs(9.81 + calibrationData[2]) > 0.7)



361        {



362            run_flag = 0;



363            printf("rsedu_control(): ERROR: Please take off from a level surface! \n");



364            return;



365        }



366        



367        if (!pressureSensorOk)



368        {



369            run_flag = 0;



370            printf("rsedu_control(): ERROR: Pressure sensor appears damaged! \n");



371            return;



372        }



373        



374        //check if optical flow up and running



375        if ((1 == MW_CONFIG_ENABLEOPTICALFLOW) && (of_fifo < 0))



376        {



377            of_fifo = open("/tmp/of_fifo", O_RDONLY); //O_NONBLOCK O_RDONLY



378            fcntl(of_fifo, F_SETFL, fcntl(of_fifo, F_GETFL) | O_NONBLOCK);



379            if (of_fifo < 0)



380            {



381                printf("rsedu_control(): ERROR optical flow not running!\n");



382                run_flag = 0;



383            }



384        }



385        



386        //Display beginning of relevant data output on screen



387        printf(" *************************************** Data_block_start *************************************************** \n");



388        



389        //Power motors



390        out->command = BLDC_CMD_RUN;



391        printf("rsedu_control(): Motors Powered. Ready for Take-Off ! \n");



392        



393        //s3: Fly



394        // ------



395    }



396    else if (counter <= onCycles)



397    {



398        if ( counter % 200 == 0)



399        {



400            uint8_t bat;



401            printf("\n **** Sensor Values when counter = %d **** \n", counter);



402            printf("rsedu_control(): sonar altitude: %fmeters, pressure: %fpascal, gyrotemp: %fcelsius\n", in->HAL_ultrasound_SI.altitude, in->HAL_pressure_SI.pressure,in->HAL_gyro_SI.temperature);



403            printf("rsedu_control(): gyro x: %frad/s, y: %frad/s, z: %frad/s\n", in->HAL_gyro_SI.x,in->HAL_gyro_SI.y,in->HAL_gyro_SI.z);



404            printf("rsedu_control(): accel x: %fm/s2, y: %fm/s2, z: %fm/s2\n", in->HAL_acc_SI.x,in->HAL_acc_SI.y,in->HAL_acc_SI.z);



405            printf("rsedu_control(): Battery output voltage Integrated: %5.2fV - %0d percents\n", in->HAL_vbat_SI.vbat_V, (int)in->HAL_vbat_SI.vbat_percentage);



406            bat = (uint8_t) in->HAL_vbat_SI.vbat_percentage;



407            write(connfd_GUI, "B", 1);



408            write(connfd_GUI, &bat, 1);



409        }



410        



411        //3.1 take off-setting



412        // +++++++++++++++++++



413        if (counter < calibCycles + takeoffCycles)



414        {



415            keyboardServerCommands->takeoff_flag = 1;  //enables take-off procedure, disables altitude-control



416            keyboardServerCommands->pos_ref[2] = -1.1;



417            



418            //React to possible low battery



419            if (in->HAL_vbat_SI.vbat_percentage < MIN_BATTTAKEOFF)



420            {



421                run_flag = 0;



422                write(connfd_GUI, "Db", 2);



423                printf("rsedu_control(): Flight aborted due to low voltage (%d %%): shutting down motors now, charge battery!\n", in->HAL_vbat_SI.vbat_percentage);



424                return;



425            }



426            



427            //3.2 transition to actual flight: enable altitude-control setting



428            // +++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++



429        }



430        else if (counter == calibCycles + takeoffCycles)



431        {



432            keyboardServerCommands->takeoff_flag = 0;  //disable take-off procedure, enables altitude-control



433            keyboardServerCommands->pos_ref[2] = -1.1;



434            



435            //3.3 actual flight setting



436            // ++++++++++++++++++++++++



437        }



438        else if (counter < onCycles)



439        {



440            



441            //Respond to requests from client



442            fcntl(connfd, F_SETFL, O_NONBLOCK);



443            memset(recvBuff, '\0', sizeof(recvBuff));



444            recv(connfd, recvBuff, sizeof(recvBuff), O_NONBLOCK);



445            processRequest(connfd,recvBuff,in,out,keyboardServerCommands, &run_flag); //  function call to process data sent from MATLAB APIs



446            



447            if (isLanding == 1)



448            {



449                // Soft land the drone upon receiving a take-off command



450                soft_land_minidrone(keyboardServerCommands, out , &run_flag);



451            }



452            



453            if ((keyboardServerCommands->orient_ref[1] == 0.0) && (keyboardServerCommands->orient_ref[2] == 0.0))



454            {



455                //control position + velocity if no specific reference attitude given (yaw angle ok)



456                keyboardServerCommands->controlModePosVSOrient = 1; //1 ; 1 position reference, 0 angle reference



457            }



458            else



459            {



460                //angle control



461                keyboardServerCommands->controlModePosVSOrient = 0; //0 ; 0 position reference, 1 angle reference



462            }



463            



464            //use of position estimate from vision



465            SensorData->VisionSensors.usePosVIS_flag = MW_CONFIG_USEIMAGEVISIONFORPOSITION;



466            



467            //React to possible Flight abort request



468            if (run_flag == 0)



469            {



470                printf("rsedu_control(): Flight abort request: shutting down motors now\n");



471                return;



472            }



473            



474            //3.4 init to stop flight (because of end of flight duration)



475            // ++++++++++++++++++++++++++++++++++++++++++++++++++++++++++



476        }



477        else



478        {



479            powerGain = 0;



480        }



481        



482        //Actual flight control



483        //---------------------



484        //safety abort for high accelerations or position



485        bool crash_detected;



486        crash_detected = (fabs(in->HAL_acc_SI.x) > MAX_ACCELERATION) || (fabs(in->HAL_acc_SI.y) > MAX_ACCELERATION);



487        



488        bool battery_low = (in->HAL_vbat_SI.vbat_percentage < MIN_BATT) && (in->HAL_vbat_SI.vbat_percentage > 1.0);



489        



490        // log any exit reasons



491        if (crash_detected)



492        {



493            printf("rsedu_control(): Flight crash detected (accelerometer): shutting down motors now. accel x: %fm/s2, y: %fm/s2, z: %fm/s2\n", in->HAL_acc_SI.x,in->HAL_acc_SI.y,in->HAL_acc_SI.z);



494        }



495        if (battery_low)



496        {



497            printf("rsedu_control(): Flight aborted due to low voltage (%d %%): shutting down motors now, charge battery!\n", in->HAL_vbat_SI.vbat_percentage);



498        }



499



500        // and stop the drone



501        if (crash_detected)



502        {



503            run_flag = 0;



504            write(connfd_GUI, "Dc", 2);



505            return;



506        }



507        if (battery_low)



508        {



509            run_flag = 0;



510            write(connfd_GUI, "Db", 2);



511            return;



512        }



513        



514        //Input to Model: optical flow computations (setup as zero-order hold: no updates on static var when nothing new in fifo)



515        SensorData->VisionSensors.opticalFlow_data[2] = 1; // Set the OF status to true. In case the below line fails (i.e when there is no data),



516                                                           // the OF error in the model increases faster. Hence, set the flag to true and the 



517                                                           // model takes the previous value. Only when data is read and the optical flow is error 



518                                                           // occurs, set the status to 'error'(-1)



519        if ((1 == MW_CONFIG_ENABLEOPTICALFLOW) && (of_fifo > 0))



520        {



521            if ((read(of_fifo, (float*)(&of_data), sizeof(of_data)) > 0) && ((of_data[0] != 0.0) || (of_data[1] != 0.0)))



522            {



523                ofQuality = of_data[3];



524                //ofDefined = of_data[4];



525                if (ofQuality > 0)



526                {



527                    counter_noOF = 0;



528                    SensorData->VisionSensors.opticalFlow_data[0] = of_data[0];



529                    SensorData->VisionSensors.opticalFlow_data[1] = of_data[1];



530                    if (counter % 200 == 0)



531                    {



532                        printf("rsedu_control(): of: %f %f %f \n",of_data[0],of_data[1],of_data[2]);



533                    }



534                }



535            }



536            else



537            {



538                 SensorData->VisionSensors.opticalFlow_data[2] = -1;  // Set the OF status to false.



539#ifndef MW_HAS_OPTICAL_FLOW_LOGIC



540                if (counter > (calibCycles + takeoffCycles + 200 * 2)) // This logic is applicable only for 18b and below models. 



541                                                                       // Wait for (calibCycles + takeoffCycles) + additional 2 seconds.



542                {



543                    counter_noOF += 1;



544                    if (counter_noOF >= MAX_noOF)



545                    {



546                        run_flag = 0;



547                        printf("rsedu_control(): Problem with optical flow, there has been no flow for %d cycles in cycle %d: shutting down motors now\n", counter_noOF, counter);



548                        return;



549                    }



550                }



551#endif



552            }



553        }



554        



555        keyboardServerCommands->live_time_ticks = counter;



556        



557#ifdef EXT_MODE



558        runModel = ((rtmGetErrorStatus(MW_StringifyDefineFunction(MODEL,_M))== (NULL)) &&



559                (!rtmGetStopRequested(MW_StringifyDefineFunction(MODEL,_M))));



560#else



561        runModel = rtmGetErrorStatus(MW_StringifyDefineFunction(MODEL,_M))== (NULL);



562#endif



563        if (runModel)



564        {



565            SensorData->HALSensors = *in;



566            #ifdef MW_HAS_IMAGE_PROCESSING



567                MW_StringifyDefineFunction(MODEL,_step0)(); /*If the Image Processing blocks are present, 



568                                                                * the step0() has the Flight Control code. */



569                



570                runVisionCode = 1; /* This flag is polled in the Vision Thread 



571                                    * to start the image processing code. 



572                                    * This flag is initially set to '0'. */



573            #else



574                MW_StringifyDefineFunction(MODEL,_step)();/*If the Image Processing blocks are not present, 



575                                                           * the step() has the Flight Control code.*/



576            #endif



577#ifdef EXT_MODE



578            rtExtModeCheckEndTrigger();



579#endif



580        }



581        else



582        {



583            runModel = 0;



584            run_flag = 0;



585#ifdef EXT_MODE



586            if (rtmGetStopRequested(MW_StringifyDefineFunction(MODEL,_M)) == true)



587            {



588                rtmSetErrorStatus(MW_StringifyDefineFunction(MODEL,_M), "Simulation finished");



589                return;



590            }



591#endif



592            write(connfd_GUI, "Dm", 2);



593            printf("rsedu_control(): ERROR: Error from simulink model @ counter=%i !\n\t%s", counter,  rtmGetErrorStatus(MW_StringifyDefineFunction(MODEL,_M)));



594            return;



595        }



596        



597        //Currently using the 'flag_outport==99' condition along with #def 'stateEstimator_MDLREF_HIDE_CHILD_' to



598        //ensure this condition is evaluated only for the demo asbQuadcopter model.



599#ifdef stateEstimator_MDLREF_HIDE_CHILD_



600        //safety abort for high mismatch OF vs state velocities



601        if (99 == flag_outport)



602        {



603            if (ofQuality > 0 &&



604                    (counter > (calibCycles + takeoffCycles)) &&



605                    (flag_outport)



606                    )



607            {



608                run_flag = 0;



609                printf("rsedu_control(): Flight crash about to happen, mismatch optical flow (%f, %f) and state estimate (%f, %f): shutting down motors now\n",



610                        20 * of_data[0],



611                        20 * of_data[1],



612                        100.0,



613                        100.1);



614                return;



615            }



616        }



617#endif



618        



619        if (flag_outport) {



620            run_flag = 0;



621            write(connfd_GUI, "F", 1);



622            write(connfd_GUI, &flag_outport, 1);



623            printf("rsedu_control(): Flight aborted due to error condition %d !\n\t",flag_outport);



624            return;



625        }



626        



627        



628        //power engines



629        if (counter < onCycles)



630        {



631            out->command = BLDC_CMD_RUN;



632        }



633        else



634        {



635            out->command = BLDC_CMD_STOP;



636            return;



637        }



638        



639        if ( counter % 200 == 0)



640        {



641            printf("rsedu_control(): motor values : %f %f %f %f\n",motors_outport[0],motors_outport[1],motors_outport[2],motors_outport[3]);



642        }



643        



644        // update motor commands with control commands



645        out->motors_speed[0] = (int)(powerGain * (fabs(motors_outport[0])));



646        out->motors_speed[1] = (int)(powerGain * (fabs(motors_outport[1])));



647        out->motors_speed[2] = (int)(powerGain * (fabs(motors_outport[2])));



648        out->motors_speed[3] = (int)(powerGain * (fabs(motors_outport[3])));



649        usleep(100);



650        



651        //s4.0 End flight: log data and close program



652        //-------------------------------------------



653    }



654    else



655    {



656        run_flag = 0;



657        stop_flight(out);



658#ifdef EXT_MODE



659        if (background_task_completed == false)



660        {



661            return;



662        }



663#endif



664#ifdef MATFILE_LOGGING



665        /* Matfile logging save*/



666        printf("rsedu_control(): Saving logged data at end of flight... \n");



667        rt_StopDataLogging(MATFILE, MW_StringifyDefineFunction(MODEL,_M)->rtwLogInfo);



668        printf("rsedu_control(): Saving logged data... DONE \n");



669#endif



670        if (1 == MW_CONFIG_ENABLEOPTICALFLOW)



671        {



672            close(of_fifo);



673        }



674        



675#if(1 == MW_CONFIG_ENABLEPROFILING)



676            fclose(ptfile);



677#endif



678        printf("rsedu_control(): Flight Completed after %i cycles ... DONE \n", counter - 1); /*Counter value is currently onCycles + 1*/



679



680#ifdef MW_HAS_IMAGE_PROCESSING



681        printf("RSEDU_image_processing: Average image processing time is %lldms \n", avrgIpTime);



682#endif



683        printf("rsedu_control(): Good night! \n");



684        write(connfd_GUI, "De", 2);



685        MW_StringifyDefineFunction(MODEL,_terminate)();



686        exit(0);



687    }



688    



689    usleep(200);



690    



691    //ptiming - store



692    //----------



693#if(1 == MW_CONFIG_ENABLEPROFILING)



694    ptimer_stopstore(start, ptfile);



695#endif



696    //----------



697}

738/** @brief Performs a running average on the sensor readings.



739 *



740 * @param[in]   ctr       The current tick counter



741 * @param[out]  calibData The calibration data to be updated



742 * @param[out]  battLevelAvg The battery level average to be updated



743 * @param[in]   in        Current sensor reading to be used to update the running average



744 */



745void perform_calibration(int ctr, real32_T* calibData, double* battLevelAvg, HAL_acquisition_t* in)



746{



747    



748    int counterCalib = ctr - 3;



749    



750    calibData[0] = calibData[0] * (counterCalib - 1) / counterCalib + in->HAL_acc_SI.x / counterCalib;



751    calibData[1] = calibData[1] * (counterCalib - 1) / counterCalib + in->HAL_acc_SI.y / counterCalib;



752    calibData[2] = calibData[2] * (counterCalib - 1) / counterCalib + in->HAL_acc_SI.z / counterCalib;



753    calibData[3] = calibData[3] * (counterCalib - 1) / counterCalib + in->HAL_gyro_SI.x / counterCalib;



754    calibData[4] = calibData[4] * (counterCalib - 1) / counterCalib + in->HAL_gyro_SI.y / counterCalib;



755    calibData[5] = calibData[5] * (counterCalib - 1) / counterCalib + in->HAL_gyro_SI.z / counterCalib;



756    calibData[6] = calibData[6] * (counterCalib - 1) / counterCalib + in->HAL_pressure_SI.pressure / counterCalib;



757    calibData[7] = calibData[7] * (counterCalib - 1) / counterCalib + in->HAL_ultrasound_SI.altitude / counterCalib;



758    



759    *battLevelAvg = *battLevelAvg * (counterCalib - 1) / counterCalib + (double)(in->HAL_vbat_SI.vbat_percentage) / counterCalib;



760}

762/**



763 * @brief Initialization of the sensor/battery calibration data



764 *



765 * @param[out]  calibData The calibration data to be initialized



766 * @param[out]  battLevelAvg The battery level average to be initialized



767 * @param[in]   in        Current sensor reading to be used to seed the running average



768 */



769void initialize_calibration_data(real32_T* calibData, double* battLevelAvg, HAL_acquisition_t* in)

773    calibData[1] = in->HAL_acc_SI.y;



774    calibData[2] = in->HAL_acc_SI.z;



775    calibData[3] = in->HAL_gyro_SI.x;

777    calibData[5] = in->HAL_gyro_SI.z;

779    calibData[7] = in->HAL_ultrasound_SI.altitude;



780    



781    *battLevelAvg = (double)((int)in->HAL_vbat_SI.vbat_percentage);



782}

784/**



785 * @brief Opens the optical flow FIFO



786 *



787 * @return RETURN_SUCCESS indicating the process was successful, RETURN_FAIL otherwise



788 */



789Return_flag_t open_optical_flow_comm(void)



790{



791    



792    int file_access_flags;



793    



794    // Initialize communication with optical flow thread



795    printf("rsedu_control(): Waiting for optical flow connection...\n");



796    usleep(100);



797    



798    of_fifo = open("/tmp/of_fifo", O_RDONLY | O_NONBLOCK);  //O_NONBLOCK O_RDONLY



799    if (of_fifo == ERROR) {



800        return RETURN_FAIL;



801    }



802    



803    file_access_flags = fcntl(of_fifo, F_GETFL);



804    if (file_access_flags == ERROR) {



805        return RETURN_FAIL;



806    }



807    



808    if (fcntl(of_fifo, F_SETFL, file_access_flags | O_NONBLOCK) == ERROR) {



809        return RETURN_FAIL;



810    }



811    



812    if (read(of_fifo, (float*)(&of_data), sizeof(of_data)) == ERROR) {



813        return RETURN_FAIL;



814    }



815    



816    return RETURN_SUCCESS;



817}
